   apTabMenu Example #4

   This example demonstrates:

  • Javascript calls. For more info see "Javascript tips"
  • Default selected tab not defined, but the toggle mode is active (pressedItem=0)
  • Tab type 2 - bold borders

   HTML Source

<applet Code=apTabMenu Archive=apTabMenu.jar Width=575 Height=22 MAYSCRIPT>
<param name="Copyright" value="Apycom Software - www.apycom.com">
<param name="bottomLine" value="false">
<param name="pressedItem" value="0">
<param name="tabType" value="2"> 
<param name="backColor" value="d0f0d0">
<param name="tabColor" value="e0ffe0">
<param name="tabHighColor" value="b0d0b0">
<param name="fontColor" value="707060">
<param name="fontHighColor" value="ffffff">
<param name="font" value="Courier,11,1">
<param name="menuItems" value="
           {Message box,javascript:1,_}
           {Confirmation box,javascript:2,_}
           {680x600 window,javascript:3,_}
           {300x200 window,javascript:4,_}
           {Launch function,javascript:5,_}
           ">
<param name="javascript:1" value="alert('Hello world!')">
<param name="javascript:2" value="confirm('Do you want to confirm?')">
<param name="javascript:3" value="open('javascript.html','_blank','width=680,height=600')">
<param name="javascript:4" value="open('javascript.html','_blank','width=300,height=200')">
<param name="javascript:5" value="my_js_function()">
</applet>


<script language="Javascript">
function my_js_function()
{
    alert('Click to continue...');
    if (confirm('Do you want to open two small windows?'))
    {
        open('javascript.html','_blank','width=150,height=150');
        open('javascript.html','_blank','width=100,height=100');
    }
    else
    {
        alert('Canceled...');
    }
}
</script>
