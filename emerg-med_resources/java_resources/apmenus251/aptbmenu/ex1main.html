This example demonstrates:

  • Tab type 3 - thin borders
  • Default selected tab is "Example #1" (pressedItem=1)
  • Using apTabMenu with frames - click any button to see how the main frame toggles
  • Item icons -
  • Click sound

HTML Source

<applet Code=apTabMenu Archive=apTabMenu.jar Width=600 Height=21>
<param name="Copyright" value="Apycom Software - www.apycom.com">
<param name="bottomLine" value="true">
<param name="pressedItem" value="1">
<param name="clickSound" value="pix/type.au">
<param name="tabType" value="3"> 
<param name="status" value="link">
<param name="backColor" value="ffffff">
<param name="tabColor" value="ffffff">
<param name="tabHighColor" value="66ccff">
<param name="fontColor" value="44aadd">
<param name="fontHighColor" value="ffffff">
<param name="font" value="Arial,11,0">
<param name="menuItems" value="
      {Example #1,ex1main.html,main_wnd,pix/app.gif}
      {Java applets,pix/ex1tst.html,main_wnd,pix/params.gif}
      {Setup,pix/ex3tst.html,main_wnd,pix/help.gif}
      {Downloads,pix/ex4tst.html,main_wnd,pix/down.gif}
      {Links,pix/ex5tst.html,main_wnd,pix/links.gif}
      {Contact,pix/ex6tst.html,main_wnd,pix/cont.gif}
      {Feedback,pix/ex7tst.html,main_wnd,pix/feed.gif}
      ">
</applet>

ex1.html

<FRAMESET rows=170,* border=0 frameBorder=0 frameSpacing=0>
<FRAME name = top_wnd scrolling=no src="ex1top.html">
<FRAME name = main_wnd scrolling=no src="ex1main.html">
</FRAMESET>
